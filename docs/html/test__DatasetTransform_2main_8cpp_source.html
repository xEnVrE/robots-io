20int main(int argc, char** argv)

31 const std::string dataset_path{argv[1]};
32
33 std::size_t skip_rows;

35 return EXIT_FAILURE;
36
37 std::size_t skip_cols;
38 if (!parse_size_t(argv, 3, "skip_cols", skip_cols))
39 return EXIT_FAILURE;
40
41 DatasetTransform dataset(dataset_path, skip_rows, skip_cols, 7);
42
43 std::size_t i = 0;
44 while (dataset.freeze())

47 auto rotation = AngleAxisd(transform.rotation());
48 std::cout << i++ << ": "

51 << std::endl;
52 }
53
54 return EXIT_SUCCESS;
